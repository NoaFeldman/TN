285 wbMatrix<wbcomplex>& TDMData::Fourier(
286  wbvector<double> &om,
287  wbMatrix<wbcomplex> &ST,
288  double alpha, double Lambda,
289  char lflag
290 ){
291 // AP comes in [nrg_iter, operator_idx, omega] \equiv [k s om] order
292  unsigned k,it,ik,is,io,l=0,no, nt=tt.len, nk=AP.dim(0), ns=AP.dim(1);
293 
294  wbArray<wbcomplex> KST(nk,ns,nt), STK;
295  wbMatrix<double> ETK(nt,nk); // exponential damping
296  wbArray<double> AR;
297  double ek,*ap,kso,o;
298  wbcomplex z,*st;
299 
300  if (lflag)
301  wblog(FL,"<i> %s (alpha=%g, Lambda=%g)", FCT, alpha, Lambda);
302 
303  getRawData(om,AR); // [ k s om ] order
304  ap=AR.data; no=om.len;
305 
306 // get damping factors (once and for all)
307  if (alpha==0) ETK.set(1); else
308  for (ik=0; ik<nk; ik++) {
309  // see also dmNRG::nrgScale()
310  ek=pow(Lambda, -double(ik)/2.) * 0.5*(Lambda+1);
311  for (it=0; it<nt; it++)
312  ETK(it,ik)=exp(-fabs(alpha*ek*tt[it]));
313  }
314 
315 // do omega summation first as it is sparse in NRG iterations
316 // KSO data -> KST
317  for (ik=0; ik<nk; ik++)
318  for (is=0; is<ns; is++)
319  for (io=0; io<no; io++, l++) if (ap[l]!=0) { kso=ap[l]; o=om[io];
320  for (it=0; it<nt; it++)
321  KST(ik,is,it)+=kso*expi(o*tt[it]);
322  }
323 
324  KST.permute(STK,"2 3 1");
325  ST.init(ns,nt);
326 
327 // convolution of NRG iterations including damping alpha
328  for (is=0; is<ns; is++)
329  for (it=0; it<nt; it++) {
330  st=STK.ref(is,it); z=0;
331  if (alpha==0)
332  for (k=0; k<nk; k++) z+=st[k];
333  else for (ap=ETK.rec(it), k=0; k<nk; k++) z+=ap[k]*st[k];
334  ST(is,it)=z;
335  }
336 
337 wblog(FL,"TST [nk ns nt no]=[%d %d %d %d]", nk, ns, nt, no);
338  ST.save2(At);
339 ST.info("ST"); At.info("At");
340 
341  return At;
342 }

423 void TDMData::getSmoothTDM(
424  wbMatrix<double> &A,
425  const wbvector<double> &omega,
426  const double eps,
427  const double sigma,
428  const char iflag
429 ){
430  unsigned i,s; // N=Ap.dim2;
431  wbvector<double> OM(N),I;
432  double x, dbl, w1=0., w2=0., om, aom;
433 
434  if (fac<=0. || sigma<=0. || eps<=0. || N<4)
435  wblog(FL,"ERR fac=%g, sigma=%g, eps=%g, N=%d ???",fac,sigma,eps,N);
436 
437  // obtain non-logarithmic (linear) omega values for regular Gaussian
438  for (i=0; i<OM.len; i++) OM[i]=idx2om(i);
439 
440  if (iflag) {
441  wblog(FL,
442  "<i> %s: 2x%d bins, %g/dec, sigma=%g",
443  FCT, N, fac/log10(M_E), sigma);
444  wblog(FL,
445  " · omega range = %.3g .. %.3g .. %.3g\n"
446  " · emin/eps/emax = %.3g .. %g .. %.3g",
447  omega.min(), omega.aMin(), omega.max(),
448  idx2om(0), eps, idx2om(N));
449  dispSumRule();
450  }
451  else {
452  wblog(FL,
453  "<i> %s: 2x%d bins\n"
454  " om: %.3g .. %.3g .. %.3g (%d)",
455  FCT, N, omega.min(), omega.aMin(), omega.max(), omega.len);
456  }
457 
458  if (Ap.isEmpty()) mergeNRGData();
459 
460  A.init(Ap.dim1, omega.len);
461 
462  initLogGaussian(N,sigma);
463 
464  for (s=0; s<Ap.dim1; s++)
465  for (i=0; i<omega.len; i++) {
466  om=omega[i]; aom=ABS(om);
467 
468  // let logGauss die out exponentially quick for |omega| < eps
469  if (aom>eps) x=1.;
470  else {
471  // quasi-loggauss in aom with maximum weight at omega=eps :D
472  if (aom==0.) x=0.;
473  else {
474  // weight x below goes down quickly, e.g. for sigma=0.5
475  // * aom = eps/10 -> x=6E-10
476  // * aom = eps/20 -> x=2E-16 i.e. about numerical noise!
477  dbl = (log(aom)-log(eps))/sigma;
478  x = exp(-dbl*dbl);
479  if (x<1E-10) x=0.;
480  }
481  }
482 
483  if (x!=0.)
484  w1 = foldLogGaussian(aom, om>0 ? Ap.rec(s):An.rec(s));
485 
486  if (x!=1.)
487  w2 = foldGaussian( om, 0.8*eps, OM.data, Ap.rec(s), N)
488  + foldGaussian(-om, 0.8*eps, OM.data, An.rec(s), N);
489 
490  A(s,i) = x*w1 + (1.-x)*w2;
491  }
492 
493  // get integrated smooth spectral function
494  Wb::dispSumRule(omega,A);
495 };

531  }

534 // (relevant only when sigma is taken too small)

632  n=b2-b1+1; ez1.init(nom,n); ez2.init(nom,n);
633  for (i=0; i<nom; i++)

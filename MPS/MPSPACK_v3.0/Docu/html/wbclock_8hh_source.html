339 double WbClock::getTickFreq() const {
340 
341  unsigned i,n=32; // test of trial runs; pick largest
342  clock_t t1,t2,nc=0;
343 
344  unsigned long long l1,l2,dl=0;
345 
346  for (i=0; i<n; i++) {
347  l1=rdtsc(); t1=clock(); while (t1==clock()) {};
348  l2=rdtsc(); t2=clock();
349 
350  if (i) {
351  nc+=(t2-t1);
352  dl+=(l2-l1);
353  }
354  // wblog(FL,"TST %2d: %6d %12d", i, t2-t1, (unsigned)(l2-l1));
355  }
356 
357 // take average
358 // nc/=(n-1); dl/=(n-1); (factors cancel out anyway)
359 
360  return (dl*(CLOCKS_PER_SEC/(double)nc));
361 }
